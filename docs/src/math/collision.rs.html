77pub fn line_line(l1: &super::Line, l2: &super::Line) -> bool {
78    fn ccw(a: &super::Point, b: &super::Point, c: &super::Point) -> bool {
79        (c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)
80    }
81    fn intersect(a: &super::Point, b: &super::Point, c: &super::Point, d: &super::Point) -> bool {
82        ccw(a, c, d) != ccw(b, c, d) && ccw(a, b, c) != ccw(a, b, d)
83    }
84    intersect(&l1.0, &l1.1, &l2.0, &l2.1)
85}
86pub fn line_closest_pt(line: &super::Line, point: &super::Point) -> super::Point {
87    let (x1, y1) = (line.0.x, line.0.y);
88    let (x2, y2) = (line.1.x, line.1.y);
89    let (x3, y3) = (point.x, point.y);
90    let (dx, dy) = (x2 - x1, y2 - y1);
91    let det = dx * dx + dy * dy;
92    let a = (dy * (y3 - y1) + dx * (x3 - x1)) / det;
93
94    let mut new_point = super::Point::new(x1 + a * dx, y1 + a * dy);
95
96    let line_length = super::get_distance(&line.0, &line.1);
97
98    let dist_to_a = super::get_distance(&new_point, &line.0);
99
100    let dist_to_b = super::get_distance(&new_point, &line.1);
101
102    if dist_to_a > line_length && dist_to_a > dist_to_b {
103        new_point = line.1;
104    }
105    if dist_to_b > line_length && dist_to_b > dist_to_a {
106        new_point = line.0;
107    }
108
109    new_point
110}

140pub fn point_rect(point: &super::Point, rect: &super::Rect) -> bool {
141    // let x1 = rect.topleft().x;
142    // let y1 = rect.topleft().y;
143    // let x2 = x1 + rect.width();
144    // let y2 = y1 + rect.height();
145    // x1 < point.x && point.x < x2 && y1 < point.y && point.y < y2
146
147    // https://gamedev.stackexchange.com/questions/128598/collision-detection-point-hitting-a-rotating-rectangle/154294#154294
148    // "Note that you don't have to rotate the rectangle.
149    // You can use the coordinates before it was rotated.
150    // You just have to rotate the point by minus the angle of the rectangle,
151    // around the center of rotation of the rectangle."
152
153    // so, as in Vupa the center of rotation of a rect is it's center, just rotate the point by -rect.angle()
154    // with rect.center() as 'origin'/'anchor'
155    // Visualisation: https://i.stack.imgur.com/vRmRH.png (Image from Heckel's response in the link above)
156    let rotated_point = super::Point::new_rotated(rect.center(), *point, -rect.rotation());
157    // super::line::rotate(super::Line::new(rect.center(), point), -rect.rotation()).1;
158
159    // and then, just check the rotated point with the non-rotated rect
160    let x1 = rect.aa_topleft().x;
161    let y1 = rect.aa_topleft().y;
162    let x2 = x1 + rect.width();
163    let y2 = y1 + rect.height();
164    x1 < rotated_point.x && rotated_point.x < x2 && y1 < rotated_point.y && rotated_point.y < y2
165
166    // false
167}
