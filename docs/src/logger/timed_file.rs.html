1use std::{
2    fs::{File, OpenOptions},
3    io::{self, Write},
4    path::PathBuf,
5};
6
7use chrono::{DateTime, Duration, Local};
8
9pub struct TimedFile {
10    interval: Duration,
11    last_update: DateTime<Local>,
12    inner_file: File,
13    path: PathBuf,
14}
15
16impl TimedFile {
17    pub fn new(path: PathBuf, interval: Duration) -> Self {
18        let datetime = Local::now();
19
20        let file = {
21            let path = Self::gen_path(path.clone(), datetime).unwrap();
22
23            match Self::open(path.clone()) {
24                Ok(file) => file, 
25                Err(why) => {
26                    eprintln!("[ERROR] Failed to create file at {path:?} due to: {why}");
27                    panic!();
28                },
29            }
30        };
31
32        Self {
33            interval,
34            last_update: datetime,
35            inner_file: file,
36            path,
37        }
38    }
39
40    fn open(path: PathBuf) -> Result<File, io::Error> {
41        OpenOptions::new().create(true).append(true).open(path)
42    }
43
44    fn gen_path(mut path: PathBuf, datetime: DateTime<Local>) -> Result<PathBuf, PathBuf> {
45        let Some(file_name) = path.file_name().and_then(|name| name.to_str()) else {
46            eprintln!("[ERROR] Failed to generate new file name based on: {:?}", path);
47            Err(path)?
48        };
49
50        let new_file_name = format!("{}-{file_name}", datetime.format("%Y-%m-%d_%H-%M-%S"));
51
52        path.set_file_name(new_file_name);
53
54        Ok(path)
55    }
56    fn update(&mut self) {
57        let now = Local::now();
58
59        if now.signed_duration_since(self.last_update) < self.interval {
60            return;
61        }
62
63        self.last_update += self.interval
64            * (now.signed_duration_since(self.last_update).num_seconds()
65                / self.interval.num_seconds()) as i32;
66
67        let Ok(new_path) = Self::gen_path(self.path.clone(), self.last_update) else {
68            return;
69        };
70
71        let new_file = match Self::open(new_path) {
72            Ok(new_file) => new_file,
73            Err(why) => {
74                log::error!("Failed to rotate TimedFile due to: {why}");
75                return;
76            }
77        };
78
79        self.inner_file = new_file;
80    }
81}
82
83impl Write for TimedFile {
84    fn write(&mut self, buf: &[u8]) -> std::io::Result<usize> {
85        self.update();
86        self.inner_file.write(buf)
87    }
88
89    fn flush(&mut self) -> std::io::Result<()> {
90        self.inner_file.flush()
91    }
92}
