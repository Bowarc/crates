15fn color(message: &str, level: &Level) -> ColoredString {
16    use colored::Colorize as _;
17    match level {
18        log::Level::Trace => message.normal(),
19        log::Level::Debug => message.cyan(),
20        log::Level::Info => message.green(),
21        log::Level::Warn => message.yellow(),
22        log::Level::Error => message.red(),
23        // _ => message.normal(),
24    }
25}

47        level: &Level,

50        format!(

52            time = chrono::Local::now().format("%H:%M:%S%.3f"),
53            level = if colored {
54                color(level.as_str(), level).to_string()
55            } else {
56                level.to_string()
57            },
58            path = source,
59            // file = file_name(file.unwrap_or("Unknown file")),
60            line_nbr = line
61                .as_ref()
62                .map(ToString::to_string)
63                .unwrap_or("?!?".to_string()),
64            message = if colored {
65                color(content, level).to_string()
66            } else {
67                content.to_string()
68            }
69        )

102fn get_most_accurate_filter(
103    source: &str,
104    filters: &[(String, LevelFilter)],
105) -> Option<log::LevelFilter> {
106    // The goal here, is to get the distance from the last position of the match to the end of the searched string
107    let get_index =
108        |s1: &str, s2: &str| -> Option<usize> { s1.find(s2).map(|i| s1.len() - s2.len() + i) };
109
110    let mut possible_filters = filters
111        .iter()
112        .flat_map(|(filter, level)| Some((level, get_index(source, filter)?)))
113        .collect::<Vec<_>>();
114
115    if possible_filters.is_empty() {
116        return None;
117    }
118
119    // Get the filter with the lowest number
120    possible_filters.sort_by_cached_key(|(_s, index)| *index);
121
122    // println!("{possible_filters:?}");
123    Some(*possible_filters.first().unwrap().0)
124}

154fn test() {
155    use log::LevelFilter;
156    let filters = vec![
157        ("Test".to_string(), LevelFilter::Off),
158        ("Test::a".to_string(), LevelFilter::Trace),
