20struct ProxyLogger {
21    #[cfg(feature = "multithread")]
22    sender: Sender<Message>,
23    #[cfg(not(feature = "multithread"))]
24    loggers: Vec<logger::Logger>,
25}
26
27impl log::Log for ProxyLogger {
28    fn enabled(&self, _metadata: &log::Metadata) -> bool {
29        true
30    }
31
32    fn log(&self, record: &log::Record) {
33        #[inline(always)]
34        fn source(path: &str, file: &str) -> String {
35            // The patern String.split(..).last().unwrap() will never panic
36
37            if file.ends_with(&format!("{}.rs", path.split("::").last().unwrap())) {
38                format!("{path}.rs")
39            } else {
40                format!(
41                    "{path}::{}",
42                    file.split('\\').last().unwrap().split('/').last().unwrap()
43                )
44            }
45        }
46
47        let line = record.line();
48        let content = record.args().to_string();
49        let level = record.level();
50        let source = source(
51            record
52                .module_path()
53                .map(ToString::to_string)
54                .as_deref()
55                .unwrap_or("Unknown module path"),
56            record
57                .file()
58                .map(ToString::to_string)
59                .as_deref()
60                .unwrap_or("Unknown file"),
61        );
62
63        #[cfg(not(feature = "multithread"))]
64        self.loggers
65            .iter()
66            .for_each(|logger| logger.log(&source, line, &content, &level));
67
68        #[cfg(feature = "multithread")]
69        if let Err(e) = self.sender.send(Message::Log {
70            source: source.clone(),
71            line,
72            content: content.clone(),
73            level,
74        }) {
75            eprintln!(
76                "[ERROR] Failed to send log record due to: {e}\n{}\n",
77                logger::Logger::gen_message(&source, line, &content, &level, false)
78            );
79        }
80    }
81
82    fn flush(&self) {
83        #[cfg(feature = "multithread")]
84        self.sender.send(Message::Flush).unwrap();
85
86        #[cfg(not(feature = "multithread"))]
87        self.loggers.iter().for_each(|logger| logger.flush());
88    }
89}

91#[cfg(feature = "multithread")]
92#[must_use]
93pub fn init(cfgs: impl Into<Vec<Config>>) -> LoggerThreadHandle {
94    let cfgs = cfgs.into();
95
96    let (sender, receiver) = mpsc::channel::<Message>();
97
98    let handle = LoggerThreadHandle::new(
99        sender.clone(),
100        std::thread::Builder::new()
101            .name("logger".to_string())
102            .spawn(move || {
103                logger(
104                    receiver,
105                    cfgs.into_iter()
106                        .map(logger::Logger::from_cfg)
107                        .collect::<Vec<logger::Logger>>(),
108                );
109            })
110            .unwrap(),
111    );
112
113    log::set_max_level(log::LevelFilter::Trace);
114    log::set_boxed_logger(Box::new(ProxyLogger { sender })).unwrap();
115
116    #[cfg(feature = "panics")]
117    log_panics::Config::new()
118        .backtrace_mode(log_panics::BacktraceMode::Resolved)
119        .install_panic_hook();
120
121    handle
122}

143#[cfg(feature = "multithread")]
144fn logger(receiver: Receiver<Message>, mut loggers: Vec<logger::Logger>) {
145    loop {
146        let message = match receiver.recv() {
147            Ok(message) => message,
148            Err(why) => {
149                eprintln!("[Logger failled on: {why}]");
150                return;
151            }
152        };
153        match message {
154            Message::Log {
155                source,
156                line,
157                content,
158                level,
159            } => {
160                loggers
161                    .iter_mut()
162                    .for_each(|logger| logger.log(&source, line, &content, &level));
163            }
164            Message::Flush => loggers.iter_mut().for_each(|logger| logger.flush()),
165            Message::Exit => break,
166        }
167    }
168}
