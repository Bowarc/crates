1use proc_macro::TokenStream;
2use quote::quote;
3use syn::{parse_macro_input, Data, DeriveInput, Error};
4
5/// A procedural macro that derives a method for enums to return the name of their variants as a string.
6///
7/// # Example
8///
9/// ```rust
10/// #[derive(enum_variant_name::VariantName)]
11/// enum MyEnum {
12///     Variant1,
13///     Variant2(()),
14///     Variant3 {
15///         _field1: (),
16///     },
17/// }
18///
19/// assert_eq!(MyEnum::Variant1.variant_name(), "Variant1");
20/// assert_eq!(MyEnum::Variant2(()).variant_name(), "Variant2");
21/// assert_eq!(MyEnum::Variant3 { _field1: () }.variant_name(), "Variant3");
22/// ```
23#[proc_macro_derive(VariantName)]
24pub fn variant_name(input: TokenStream) -> TokenStream {
25    let input = parse_macro_input!(input as DeriveInput);
26
27    let Data::Enum(enum_data) = &input.data else {
28        return Error::new_spanned(&input, "VariantName can only be implemented for enums")
29            .to_compile_error()
30            .into();
31    };
32
33    let match_arms = enum_data.variants.iter().map(|variant| {
34        let variant_ident = &variant.ident;
35        let variant_string = &variant_ident.to_string();
36        let fields_in_variant = quote! { {..} };
37        quote! {
38            Self::#variant_ident #fields_in_variant => #variant_string,
39        }
40    });
41
42    let match_stmnt = quote! {
43        match self {
44            #(#match_arms)*
45        }
46    };
47
48    let name = &input.ident;
49
50    let (impl_generics, ty_generics, where_clause) = &input.generics.split_for_impl();
51
52    TokenStream::from(quote! {
53        impl #impl_generics #name #ty_generics #where_clause {
54            pub fn variant_name(&self) -> &'static str {
55                #match_stmnt
56           }
57        }
58    })
59}
