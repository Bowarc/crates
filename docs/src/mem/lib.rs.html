1pub enum Prefix {
2    Decimal,
3    Binary,
4}
5
6impl Prefix{
7    const DECIMAL_UNITS: &[&str] = &["b", "Kb", "Mb", "Gb", "Tb", "Pb", "Eb", "Zb", "Yb"];
8    const BINARY_UNITS: &[&str] = &["b", "Kib", "Mib", "Gib", "Tib", "Pib", "Eib", "Zib", "Yib"];
9
10    const fn units(&self) -> &[&str]{
11        match self{
12            Self::Decimal => Self::DECIMAL_UNITS,
13            Self::Binary => Self::BINARY_UNITS,
14        }
15    }
16}
17
18impl From<&Prefix> for f64 {
19    fn from(prefix: &Prefix) -> f64 {
20        match prefix {
21            Prefix::Decimal => 1000.,
22            Prefix::Binary => 1024.,
23        }
24    }
25}

27pub fn format(bytes: u64, prefix: &Prefix) -> String {
28    if bytes == 0 {
29        return "0 B".to_string();
30    }
31    
32    let units = prefix.units();
33    let prefix = f64::from(prefix);
34    
35    let mut size = bytes as f64;
36    let mut index = 0;
37
38    while size >= prefix && index < units.len() - 1 {
39        size /= prefix;
40        index += 1;
41    }
42
43    format!("{:.2} {}", size, units[index])
44}

48    format(bytes, &Prefix::Decimal)

51#[inline(always)]
52pub fn format_binary(bytes: u64) -> String{
